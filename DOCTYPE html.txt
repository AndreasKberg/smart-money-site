    document.getElementById('year').textContent = new Date().getFullYear();

    function handleForm(e) {
      e.preventDefault();
      const form = e.target;
      const name = encodeURIComponent(form.name.value.trim());
      const email = encodeURIComponent(form.email.value.trim());
      const message = encodeURIComponent(form.message.value.trim());
      const subject = `Smart Money Smarts: Message from ${form.name.value.trim()}`;
      const body = 
        `Name: ${form.name.value.trim()}\\n` +
        `Email: ${form.email.value.trim()}\\n\\n` +
        `Message:\\n${form.message.value.trim()}`;
      const mailto = `mailto:smartmoney@yourdomain.com?subject=${subject}&body=${body}`;

      document.getElementById('formResult').style.display = 'block';
      setTimeout(() => {
        window.location.href = mailto;
      }, 300);
    }

    // Smooth scroll for nav links
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
      anchor.addEventListener('click', function (e) {
        e.preventDefault();
        const target = document.querySelector(this.getAttribute('href'));
        if (target) {
          target.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }
      });
    });
